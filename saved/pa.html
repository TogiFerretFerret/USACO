#include <bits/stdc++.h>
using namespace std;

template <class T> using V = vector<T>;

int main() {
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	int N;
	cin >> N;
	V<V<int>> routes(N, V<int>(N));
	for (int i = 0; i < N - 1; ++i) {
		string s;
		cin >> s;
		for (int j = i + 1; j < N; ++j) routes[i][j] = s.at(j - i - 1) - '0';
	}
	V<V<int>> direct(N, V<int>(N));
	auto count_routes_excluding_direct = [&](int l, int r) {
		V<int> routes_to(N);
		routes_to[l] = 1;
		for (int i = l + 1; i <= r; ++i)
			for (int j = l; j < i; ++j)
				routes_to[i] ^= routes_to[j] * direct[j][i];
		return routes_to[r];
	};
	int ans = 0;
	for (int i = N - 1; i >= 0; --i)
		for (int j = i + 1; j < N; ++j) {
			direct[i][j] = routes[i][j] ^ count_routes_excluding_direct(i, j);
			ans += direct[i][j];
		}
	cout << ans << "\n";
}
